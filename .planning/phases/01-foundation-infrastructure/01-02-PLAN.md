**1. Replace `src/app/layout.tsx`** with HTML-shell-only root layout. Keep Geist font imports and globals.css. Remove the `metadata` export (metadata moves to locale layout). Do NOT include locale-specific logic here. The `lang` attribute will NOT be set here -- it moves to the locale layout's parent or is handled by the locale segment:

```typescript
import { Geist, Geist_Mono } from 'next/font/google'
import './globals.css'

const geistSans = Geist({
  variable: '--font-geist-sans',
  subsets: ['latin'],
})

const geistMono = Geist_Mono({
  variable: '--font-geist-mono',
  subsets: ['latin'],
})

export default function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode
}>) {
  return (
    <html className={`${geistSans.variable} ${geistMono.variable} h-full antialiased`}>
      <body className="min-h-full flex flex-col">{children}</body>
    </html>
  )
}
```

NOTE: The `lang` attribute is intentionally omitted from the root `<html>` tag. It should be set dynamically in the locale layout. However, since root layout renders `<html>`, and locale layout is a child, the locale layout cannot set `lang` on `<html>`. There are two approaches:
- Accept that root layout sets `lang="en"` as default and rely on hreflang for SEO (simpler)
- Pass locale up via a shared mechanism

For Phase 1, set `lang="en"` as the default. This is acceptable because search engines use hreflang tags (Phase 2) as the primary locale signal, not the `lang` attribute.

**3. Create `src/app/[locale]/layout.tsx`** -- the locale-aware layout with AdSense placeholders, header with site name + language switcher, and NextIntlClientProvider:

```typescript
import { NextIntlClientProvider, hasLocale } from 'next-intl'
import { setRequestLocale, getTranslations } from 'next-intl/server'
import { notFound } from 'next/navigation'
import { routing } from '@/i18n/routing'

export function generateStaticParams() {
  return routing.locales.map((locale) => ({ locale }))
}

export default async function LocaleLayout({
  children,
  params,
}: {
  children: React.ReactNode
  params: Promise<{ locale: string }>
}) {
  const { locale } = await params

  if (!hasLocale(routing.locales, locale)) {
    notFound()
  }

  setRequestLocale(locale)

  const t = await getTranslations({ locale, namespace: 'common' })
  const tLayout = await getTranslations({ locale, namespace: 'layout' })

  const otherLocale = locale === 'en' ? 'vi' : 'en'
  const switchHref = locale === 'en' ? '/vi/' : '/'

  return (
    <NextIntlClientProvider>
      {/* Header with site name and language switcher */}
      <header className="w-full border-b border-zinc-200 dark:border-zinc-800">
        {/* Header ad slot: 90px min-height */}
        <div
          className="min-h-[90px] w-full bg-zinc-50 dark:bg-zinc-900 flex items-center justify-center text-xs text-zinc-400"
          aria-hidden="true"
          data-ad-slot="header"
        >
          {tLayout('adPlaceholder')}
        </div>
        <div className="mx-auto max-w-6xl flex items-center justify-between px-4 py-3">
          <a href={locale === 'en' ? '/' : '/vi/'} className="text-lg font-semibold text-zinc-900 dark:text-zinc-100">
            {t('siteName')}
          </a>
          <a
            href={switchHref}
            className="text-sm px-3 py-1 rounded border border-zinc-300 dark:border-zinc-700 text-zinc-600 dark:text-zinc-400 hover:bg-zinc-100 dark:hover:bg-zinc-800 transition-colors"
          >
            {t(`languageSwitcher.${otherLocale}`)}
          </a>
        </div>
      </header>

      {/* Main content area with sidebar */}
      <div className="mx-auto max-w-6xl flex flex-1 px-4 py-6 gap-6">
        {/* Main content column */}
        <main className="flex-1 min-w-0">
          {children}

          {/* Below-tool ad slot: 90px min-height */}
          <div
            className="mt-6 min-h-[90px] w-full bg-zinc-50 dark:bg-zinc-900 flex items-center justify-center text-xs text-zinc-400"
            aria-hidden="true"
            data-ad-slot="below-tool"
          >
            {tLayout('adPlaceholder')}
          </div>
        </main>

        {/* Sidebar column -- hidden on mobile */}
        <aside className="hidden lg:block w-72 shrink-0">
          {/* Sidebar ad slot: 250px min-height */}
          <div
            className="min-h-[250px] w-full bg-zinc-50 dark:bg-zinc-900 flex items-center justify-center text-xs text-zinc-400 sticky top-6"
            aria-hidden="true"
            data-ad-slot="sidebar"
          >
            {tLayout('adPlaceholder')}
          </div>
        </aside>
      </div>
    </NextIntlClientProvider>
  )
}
```

Key requirements:
- `generateStaticParams` returns `[{locale: 'en'}, {locale: 'vi'}]` -- child pages inherit this
- `setRequestLocale(locale)` called BEFORE any `getTranslations` or other next-intl hooks (CRITICAL for SSG)
- `NextIntlClientProvider` wraps children with NO manual `messages`/`locale` props (v4 auto-inherits)
- Three AdSense placeholder divs with `data-ad-slot` attributes and correct min-heights: header=90px, sidebar=250px, below-tool=90px
- Sidebar has `hidden lg:block` to collapse on mobile
- Sidebar has `sticky top-6` to stick to viewport top
- Header includes site name (links to locale root) and language switcher (links to other locale's root)

**5. Create `src/app/[locale]/[tool]/page.tsx`** -- tool page placeholder with generateStaticParams:

```typescript
import { setRequestLocale, getTranslations } from 'next-intl/server'
import { getToolSlugs, getToolBySlug } from '@/lib/tools'
import { notFound } from 'next/navigation'

export function generateStaticParams() {
  return getToolSlugs().map((tool) => ({ tool }))
}

export default async function ToolPage({
  params,
}: {
  params: Promise<{ locale: string; tool: string }>
}) {
  const { locale, tool: toolSlug } = await params
  setRequestLocale(locale)

  const tool = getToolBySlug(toolSlug)
  if (!tool) {
    notFound()
  }

  const t = await getTranslations({ locale, namespace: tool.i18nKey })

  return (
    <div className="py-8">
      <h1 className="text-3xl font-bold text-zinc-900 dark:text-zinc-100">
        {t('name')}
      </h1>
      <p className="mt-2 text-lg text-zinc-600 dark:text-zinc-400">
        {t('description')}
      </p>
      <p className="mt-8 text-sm text-zinc-500">
        Tool UI coming in Phase 3.
      </p>
    </div>
  )
}
```

`generateStaticParams` returns slugs for non-homepage tools only (case-converter is served at `/` by the homepage page.tsx, not via `[tool]`). This combines with the parent `[locale]/layout.tsx` generateStaticParams to produce all locale+tool combinations.
