     const requested = await requestLocale
     const locale = hasLocale(routing.locales, requested)

       : routing.defaultLocale

     return {
       locale,
       messages: (await import(`../../messages/${locale}.json`)).default,

```typescript
export interface Tool {
  slug: string
  i18nKey: string
  category: string
  relatedSlugs: string[]
  isHomepage?: boolean
}

export const tools: Tool[] = [
  {
    slug: 'case-converter',
    i18nKey: 'tools.caseConverter',
    category: 'text',
    relatedSlugs: ['reverse-text', 'slug-generator'],
    isHomepage: true,
  },
  {
    slug: 'reverse-text',
    i18nKey: 'tools.reverseText',
    category: 'text',
    relatedSlugs: ['case-converter', 'slug-generator'],
  },
  {
    slug: 'base64-encode-decode',
    i18nKey: 'tools.base64',
    category: 'encoding',
    relatedSlugs: [],
  },
  {
    slug: 'slug-generator',
    i18nKey: 'tools.slugGenerator',
    category: 'text',
    relatedSlugs: ['case-converter', 'reverse-text'],
  },
  {
    slug: 'password-generator',
    i18nKey: 'tools.passwordGenerator',
    category: 'generator',
    relatedSlugs: [],
  },
]

/** Get the homepage tool (case-converter) */
export function getHomepageTool(): Tool {
  return tools.find((t) => t.isHomepage)!
}

/** Get a tool by slug */
export function getToolBySlug(slug: string): Tool | undefined {
  return tools.find((t) => t.slug === slug)
}

/** Get all non-homepage tool slugs (for [tool] dynamic route) */
export function getToolSlugs(): string[] {
  return tools.filter((t) => !t.isHomepage).map((t) => t.slug)
}
```

The registry carries: `slug` (URL segment), `i18nKey` (translation key prefix), `category` (grouping), `relatedSlugs` (for related-tools widget in Phase 3), `isHomepage` (true only for case-converter).

5 tools seeded: case-converter, reverse-text, base64-encode-decode, slug-generator, password-generator.

Helper functions provided for common lookups: `getHomepageTool()`, `getToolBySlug()`, `getToolSlugs()`.
