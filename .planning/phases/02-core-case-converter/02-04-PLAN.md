Create `src/components/JsonLd.tsx` as a Server Component (no 'use client'). Renders inline `<script type="application/ld+json">`.

```typescript
import { getTranslations } from 'next-intl/server'

interface JsonLdProps {
  locale: string
}

export default async function JsonLd({ locale }: JsonLdProps) {
  const t = await getTranslations({ locale, namespace: 'seo' })
  const tTool = await getTranslations({ locale, namespace: 'tools' })

  const baseUrl = 'https://text-case-converter-chi.vercel.app'
  const pageUrl = locale === 'en' ? baseUrl + '/' : baseUrl + '/vi/'
  const howToSteps = t.raw('howToSteps') as string[]

  const jsonLd = {
    '@context': 'https://schema.org',
    '@graph': [
      {
        '@type': 'SoftwareApplication',
        name: tTool('caseConverter.name'),
        description: t('description'),
        url: pageUrl,
        applicationCategory: 'UtilityApplication',
        operatingSystem: 'Any',
        offers: {
          '@type': 'Offer',
          price: '0',
          priceCurrency: 'USD',
        },
        inLanguage: locale,
      },
      {
        '@type': 'HowTo',
        name: t('howToName'),
        step: howToSteps.map((text, i) => ({
          '@type': 'HowToStep',
          position: i + 1,
          text,
        })),
      },
    ],
  }

  return (
    <script
      type="application/ld+json"
      dangerouslySetInnerHTML={{ __html: JSON.stringify(jsonLd) }}
    />
  )
}
```

**Key details:**
- Use `@graph` array to include both SoftwareApplication and HowTo in a single script tag
- `dangerouslySetInnerHTML` with `JSON.stringify` — no `next/script` (per UI-SPEC)
- `applicationCategory: 'UtilityApplication'` for Google Rich Results
- `offers` with price 0 to indicate free tool
- `inLanguage` set to the current locale
- HowTo steps use `position` (1-indexed) per schema.org spec
- Base URL is the Vercel deployment URL from Phase 1

import type { Metadata } from 'next'
import { getTranslations } from 'next-intl/server'

type Props = {
  params: Promise<{ locale: string }>
}

export async function generateMetadata({ params }: Props): Promise<Metadata> {
  const { locale } = await params
  const t = await getTranslations({ locale, namespace: 'seo' })

  const baseUrl = 'https://text-case-converter-chi.vercel.app'
  const canonicalUrl = locale === 'en' ? baseUrl + '/' : baseUrl + '/vi/'

  return {
    title: t('title'),
    description: t('description'),
    alternates: {
      canonical: canonicalUrl,
      languages: {
        en: baseUrl + '/',
        vi: baseUrl + '/vi/',
      },
    },
    openGraph: {
      title: t('title'),
      description: t('description'),
      url: canonicalUrl,
      siteName: 'Text Case Converter',
      images: [
        {
          url: baseUrl + '/og/text-case-converter.png',
          width: 1200,
          height: 630,
          alt: t('title'),
        },
      ],
      locale: locale === 'en' ? 'en_US' : 'vi_VN',
      type: 'website',
    },
    twitter: {
      card: 'summary_large_image',
      title: t('title'),
      description: t('description'),
      images: [baseUrl + '/og/text-case-converter.png'],
    },
  }
}
